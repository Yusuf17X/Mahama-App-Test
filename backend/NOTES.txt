function checkBadgeEarned(user, badge) {
  switch (badge.requirement_type) {
    case 'challenges_count':
      const completedCount = await getCompletedChallengesCount(user.id)
      return completedCount >= badge.requirement_value
    
    case 'points_threshold':
      return user.points >= badge.requirement_value
    
    case 'specific_challenge':
      const hasCompleted = await hasCompletedChallenge(user.id, badge.requirement_value)
      return hasCompleted
    
    default:
      return false
  }
}
